comprehensions.py
# === LIST COMPREHENSION ===
# Syntax: [expression for item in iterable if condition]

# Normal way:
squares = []
for x in range(10):
    squares.append(x**2)

# Comprehension way (ek line mein!):
squares = [x**2 for x in range(10)]

# With condition:
even_sq = [x**2 for x in range(10) if x % 2 == 0]

# Practical example:
names = ["alice", "bob", "charlie"]
upper_names = [n.upper() for n in names if len(n) > 3]

# === DICT COMPREHENSION ===
nums = [1, 2, 3, 4]
sq_dict = {n: n**2 for n in nums}    # {1:1, 2:4, 3:9, 4:16}

# Reverse a dict:
original = {"a": 1, "b": 2}
reversed_d = {v: k for k, v in original.items()}  # {1:'a', 2:'b'}

# === SET COMPREHENSION ===
data = [1, 2, 2, 3, 3, 4]
unique_sq = {x**2 for x in data}   # {1, 4, 9, 16}

# === GENERATOR EXPRESSION ===
# List jaisa but memory efficient (lazy evaluation)
gen = (x**2 for x in range(1000000))  # memory nahi khata!
print(next(gen))  # 0
print(sum(gen))   # sab sum karo — efficient!

# --- Class Define Karna ---
class Employee:
    # Class variable (sab objects share karte hain)
    company = "TechCorp"

    def __init__(self, name, dept, salary):
        # Instance variables (har object ka alag)
        self.name = name
        self.dept = dept
        self.salary = salary

    def introduce(self):
        return f"Main {self.name}, {self.dept} mein kaam karta hoon."

    def give_raise(self, amount):
        self.salary += amount
        return f"New salary: ₹{self.salary}"

    def __str__(self):                       # print ke liye
        return f"Employee({self.name}, {self.dept})"

    def __repr__(self):                      # debugging ke liye
        return f"Employee(name={self.name!r})"

# --- Object Banana ---
emp1 = Employee("Rahul", "IT", 55000)
emp2 = Employee("Priya", "HR", 48000)

print(emp1.introduce())
print(emp1.give_raise(5000))

print(emp1)                # __str__ use hoga
🧬 Inheritance — Parent se Inherit Karo
Code Reuse
inheritance.py
# --- Parent Class ---
class Animal:
    def __init__(self, name, sound):
        self.name = name
        self.sound = sound

    def speak(self):
        return f"{self.name} says {self.sound}"

# --- Child Class (inherits Animal) ---
class Dog(Animal):
    def __init__(self, name, breed):
        super().__init__(name, "Woof")    # parent ka __init__
        self.breed = breed

    def fetch(self):                      # naya method
        return f"{self.name} is fetching!"

    def speak(self):                      # Override!
        return f"{self.name} barks: WOOF WOOF!"

dog = Dog("Bruno", "Labrador")
print(dog.speak())    # Dog's version
print(dog.fetch())    # Dog's own method

# --- isinstance check ---
print(isinstance(dog, Dog))     # True
print(isinstance(dog, Animal))  # True (parent bhi)

# --- @property, @staticmethod, @classmethod ---
class Circle:
    PI = 3.14159

    def __init__(self, radius):
        self._radius = radius

    @property
    def area(self):                  # method ko attribute jaisa use karo
        return Circle.PI * self._radius ** 2

    @staticmethod
    def is_valid(r):                 # self nahi chahiye
        return r > 0

    @classmethod
    def unit_circle(cls):           # class se object banana
        return cls(1)

c = Circle(5)
print(c.area)              # 78.53... (no brackets needed)
print(Circle.is_valid(3))  # True

🎭 Decorators
Function Wrap Karna

Decorator ek function hai jo doosre function ko modify karta hai bina uska code chhue. @syntax use hoti hai.

decorators.py
import time

# --- Simple Decorator ---
def timer(func):
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"{func.__name__} took {end-start:.4f}s")
        return result
    return wrapper

@timer                         # = slow_func = timer(slow_func)
def slow_func():
    time.sleep(1)
    return "Done"

slow_func()   # automatically time count hoga

# --- Practical: Login Check ---
def require_login(func):
    def wrapper(user, *args, **kwargs):
        if not user.get("logged_in"):
            return "Login karein pehle!"
        return func(user, *args, **kwargs)
    return wrapper

@require_login
def dashboard(user):
    return f"Welcome, {user['name']}!"
🔄 Generators — Memory Efficient Loops
Large Data

Generator ek function hai jo values ek-ek karke yield karta hai — sab ek saath memory mein nahi. Large files ya SQL results ke liye perfect.

generators.py
# --- Generator Function ---
def count_up(n):
    for i in range(n):
        yield i       # return nahi, yield

gen = count_up(5)
print(next(gen))   # 0
print(next(gen))   # 1

for val in count_up(5):
    print(val)

# --- Practical: Large File Processing ---
def read_large_file(filepath):
    with open(filepath) as f:
        for line in f:
            yield line.strip()  # ek line at a time!

# RAM exhaust nahi hoga, chahe file 10GB ho
for line in read_large_file("huge_data.txt"):
    print(line)

🛡️ try / except / finally
Production Code
⚠️
Real programs mein try-except ZARURI hai. SQL queries, file ops, user input — sab mein errors aa sakte hain.
error_handling.py
# --- Basic try-except ---
try:
    num = int(input("Number do: "))
    result = 10 / num
    print(result)
except ValueError:
    print("Number nahi diya!")
except ZeroDivisionError:
    print("Zero se divide nahi kar sakte!")
except Exception as e:
    print(f"Koi error: {e}")   # koi bhi error
else:
    print("Sab theek raha!")    # error nahi aya tab
finally:
    print("Ye hamesha chalega")  # error ho ya na ho

# --- Common Exceptions ---
# ValueError    — wrong type ka value
# TypeError     — wrong type ka operation
# KeyError      — dict mein key nahi
# IndexError    — list index out of range
# FileNotFoundError — file nahi mili
# ZeroDivisionError — 0 se divide

# --- Custom Exception ---
class AgeError(Exception):
    def __init__(self, age):
        super().__init__(f"Invalid age: {age}")

def validate_age(age):
    if age < 0 or age > 150:
        raise AgeError(age)
    return True

try:
    validate_age(-5)
except AgeError as e:
    print(e)   # "Invalid age: -5"

with sqlite3.connect("company.db") as conn:
    cur = conn.cursor()

    # --- DELETE ---
    cur.execute("DELETE FROM employees WHERE id = ?", (5,))
    cur.execute("DELETE FROM employees WHERE salary < ?", (30000,))
    cur.execute("DELETE FROM employees")       # sab delete!

    # === TABLE JOIN ===
    # 2 tables banao pehle
    cur.execute("""CREATE TABLE IF NOT EXISTS departments (
        dept_id   INTEGER PRIMARY KEY,
        dept_name TEXT,
        manager   TEXT,
        budget    REAL
    )""")

    # INNER JOIN — dono mein match karne wale
    cur.execute("""
        SELECT e.name, e.salary, d.manager, d.budget
        FROM employees e
        INNER JOIN departments d ON e.dept = d.dept_name
        WHERE e.salary > 50000
        ORDER BY e.salary DESC
    """)

    # LEFT JOIN — employees sab aayenge
    cur.execute("""
        SELECT e.name, e.dept, d.budget
        FROM employees e
        LEFT JOIN departments d ON e.dept = d.dept_name
    """)

    # === SUBQUERY ===
    # Average se zyada salary wale
    cur.execute("""
        SELECT name, salary FROM employees
        WHERE salary > (SELECT AVG(salary) FROM employees)
    """)

    # === TRANSACTION (Ek saath sab ho ya kuch nahi) ===
    try:
        conn.execute("BEGIN")
        cur.execute("UPDATE accounts SET balance = balance - 1000 WHERE id = 1")
        cur.execute("UPDATE accounts SET balance = balance + 1000 WHERE id = 2")
        conn.commit()    # dono saath save honge
    except Exception as e:
        conn.rollback()  # error aya? sab wapas!
        print(f"Transaction failed: {e}")

import sqlite3

class EmployeeDB:
    def __init__(self, db_name="company.db"):
        self.conn = sqlite3.connect(db_name)
        self.conn.row_factory = sqlite3.Row
        self._create_table()

    def _create_table(self):
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS employees (
                id      INTEGER PRIMARY KEY AUTOINCREMENT,
                name    TEXT NOT NULL,
                dept    TEXT,
                salary  REAL DEFAULT 0
            )""")
        self.conn.commit()

    def add(self, name, dept, salary):
        cur = self.conn.execute(
            "INSERT INTO employees (name, dept, salary) VALUES (?,?,?)",
            (name, dept, salary)
        )
        self.conn.commit()
        return cur.lastrowid

    def get_all(self):
        cur = self.conn.execute("SELECT * FROM employees ORDER BY salary DESC")
        return [dict(r) for r in cur.fetchall()]

    def get_by_dept(self, dept):
        cur = self.conn.execute(
            "SELECT * FROM employees WHERE dept = ?", (dept,))
        return [dict(r) for r in cur.fetchall()]

    def update_salary(self, emp_id, new_salary):
        self.conn.execute(
            "UPDATE employees SET salary = ? WHERE id = ?",
            (new_salary, emp_id)
        )
        self.conn.commit()

    def delete(self, emp_id):
        self.conn.execute("DELETE FROM employees WHERE id = ?", (emp_id,))
        self.conn.commit()

    def dept_stats(self):
        cur = self.conn.execute("""
            SELECT dept,
                   COUNT(*) as headcount,
                   AVG(salary) as avg_salary,
                   MAX(salary) as max_salary
            FROM employees
            GROUP BY dept
            ORDER BY avg_salary DESC
        """)
        return [dict(r) for r in cur.fetchall()]

    def search(self, keyword):
        cur = self.conn.execute(
            "SELECT * FROM employees WHERE name LIKE ?",
            (f"%{keyword}%",)
        )
        return [dict(r) for r in cur.fetchall()]

    def close(self):
        self.conn.close()

# --- Usage ---
db = EmployeeDB()
db.add("Rahul", "IT", 55000)
db.add("Priya", "HR", 48000)
db.add("Arjun", "IT", 67000)

print("All employees:", db.get_all())
print("IT dept:", db.get_by_dept("IT"))
print("Dept stats:", db.dept_stats())
db.update_salary(1, 60000)
db.close()

❓ Jab Bhool Jao — Quick Answers

list vs tuple vs set vs dict kab?

→ list: order chahiye + change karna ho. [1,2,3]

→ tuple: fixed data, coordinates, function return. (1,2)

→ set: unique values chahiye, duplicates hatane. {1,2,3}

→ dict: naam se access karna ho. {"key": value}

*args aur **kwargs ka difference?

→ *args: variable number of positional arguments → tuple mein milte hain

→ **kwargs: variable number of keyword arguments → dict mein milte hain

Example: func(1,2,3) → args = (1,2,3) | func(a=1,b=2) → kwargs = {'a':1,'b':2}

SQL mein ? kyu use karte hain, f-string kyu nahi?

→ f-string se SQL Injection attack ho sakta hai. Koi bhi user malicious code daal sakta hai.

→ ❌ f"WHERE name = '{user_input}'" — DANGEROUS

→ ✅ "WHERE name = ?", (user_input,) — SAFE

commit() kyu zaruri hai?

→ INSERT/UPDATE/DELETE ke baad changes RAM mein hote hain, disk pe nahi.

→ conn.commit() permanently save karta hai.

→ with block use karo — automatically commit hoga.

fetchone() vs fetchall() vs fetchmany()?

→ fetchone(): sirf ek row. None agar koi nahi.

→ fetchall(): sab rows list mein. Empty list agar koi nahi.

→ fetchmany(n): n rows. Large datasets ke liye paginate karo.
